## 改进背景与目标

### 当前问题分析

| 问题类型 | 具体表现 | 影响 | 状态 |
|---------|---------|------|------|
| **证据可靠性缺失** | 过时标签、AI提取误差与教师确认同等权重 | 不可靠证据可能主导推断 | 待解决 |
| **证据冲突未消解** | 风险证据与保护性证据简单累乘 | 矛盾证据相互抵消，无法体现保护性因素价值 | 待解决 |
| **维度独立性假设** | 家庭→情绪→社交的因果链未建模 | 关联风险传导被忽略 | ✅ **已由传导层方案解决** |
| **缺失证据误判** | 数据缺失等同于"无证据" | 置信度虚高，决策风险增加 | 待解决 |
| **似然比固化** | 所有证据的LR值静态配置 | 无法从教师反馈中学习优化 | 待解决 |
| **点估计输出** | 仅输出单一概率值 | 无法量化不确定性 | 待解决 |
| **异常证据未检测** | 矛盾证据不触发人工确认 | 数据质量问题被掩盖 | 待解决 |

### B1. 证据可靠性评分机制

#### 问题分析

当前系统中，所有证据的似然比直接参与累乘，不考虑证据来源的可靠性差异：

```
教师确认已解决 (LR=0.5, 高可靠性)
  vs
过时的家庭困难标签 (LR=1.8, 低可靠性)

当前处理：同等权重累乘
理想处理：高可靠性证据权重更高
```

**2. 实现可靠性调整函数**

```python
# 修改文件：backend/services/student_care_bayes_service.py

import math
from core.student_care_evidence_reliability import EVIDENCE_RELIABILITY_SCORES, DEFAULT_RELIABILITY

def adjust_lr_by_reliability(lr: float, evidence_key: str) -> float:
    """
    根据证据可靠性调整似然比
    
    数学原理：
    - 似然比代表证据对风险的支持程度
    - 可靠性代表证据本身的可信程度
    - 有效似然比 = LR^reliability
    
    效果：
    - 高可靠性证据（reliability=0.95）：LR保持接近原值
      例：LR=2.0 → 有效LR = 2.0^0.95 = 1.93
    - 低可靠性证据（reliability=0.50）：LR被压缩
      例：LR=2.0 → 有效LR = 2.0^0.50 = 1.41
    
    Args:
        lr: 原始似然比
        evidence_key: 证据标识符
    
    Returns:
        float: 调整后的有效似然比
    
    Example:
        >>> adjust_lr_by_reliability(3.0, "teacher_review_resolved")
        2.86  # 高可靠性，几乎不变
        
        >>> adjust_lr_by_reliability(3.0, "tag_family_hardship")
        1.73  # 低可靠性，大幅压缩
    """
    # 获取证据可靠性分数
    reliability = EVIDENCE_RELIABILITY_SCORES.get(evidence_key, DEFAULT_RELIABILITY)
    
    # 应用可靠性调整
    # LR^reliability 使得低可靠性证据的影响被压缩
    if lr >= 1.0:
        # 风险证据：压缩过度乐观的估计
        adjusted_lr = lr ** reliability
    else:
        # 保护性证据（LR<1）：压缩过度乐观的保护效果
        # LR=0.5, reliability=0.5 → 0.5^0.5 = 0.71（保护效果减弱）
        adjusted_lr = lr ** reliability
    
    # 约束到合理范围
    adjusted_lr = max(0.1, min(adjusted_lr, 10.0))
    
    return adjusted_lr


def batch_adjust_lrs_by_reliability(
    evidence_list: list[dict]
) -> list[dict]:
    """
    批量调整证据列表的似然比
    
    Args:
        evidence_list: 证据列表，每个元素包含 {
            "key": str,           # 证据标识
            "lr": float,          # 原始似然比
            "source": str,        # 证据来源
            "description": str    # 证据描述
        }
    
    Returns:
        list[dict]: 调整后的证据列表，新增字段 {
            "original_lr": float,     # 原始似然比
            "reliability": float,     # 可靠性分数
            "effective_lr": float     # 有效似然比
        }
    """
    adjusted_list = []
    
    for evidence in evidence_list:
        evidence_key = evidence["key"]
        original_lr = evidence["lr"]
        reliability = EVIDENCE_RELIABILITY_SCORES.get(evidence_key, DEFAULT_RELIABILITY)
        effective_lr = adjust_lr_by_reliability(original_lr, evidence_key)
        
        adjusted_evidence = {
            **evidence,
            "original_lr": original_lr,
            "reliability": reliability,
            "effective_lr": effective_lr,
        }
        adjusted_list.append(adjusted_evidence)
    
    return adjusted_list
```

**3. 集成到贝叶斯计算流程**

```python
# 修改文件：backend/services/student_care_bayes_service.py

def _calculate_posterior_for_dimension(
    dimension: str,
    base_prior: float,
    linear_score: float,
    evidence_list: list[dict],
    profile: StudentCareProfile = None,
) -> dict:
    """
    计算单个维度的后验概率（集成可靠性调整）
    
    流程：
    1. 计算动态先验（基于线性分数）
    2. 提取证据似然比
    3. 【新增】应用可靠性调整
    4. 应用边际递减
    5. Odds-LR累乘
    6. 返回后验概率及详细信息
    
    Returns:
        {
            "posterior": float,           # 后验概率
            "prior": float,               # 先验概率
            "dynamic_prior": float,       # 动态先验
            "evidence_count": int,        # 证据数量
            "effective_lrs": list,        # 有效似然比列表
            "reliability_adjusted": bool, # 是否应用了可靠性调整
            "evidence_details": list,     # 证据详情（含可靠性）
        }
    """
    # Step 1: 计算动态先验
    dynamic_prior = _calculate_dynamic_prior(base_prior, linear_score)
    
    # Step 2: 提取原始似然比
    original_lrs = [e["lr"] for e in evidence_list]
    
    # Step 3: 应用可靠性调整
    adjusted_evidence = batch_adjust_lrs_by_reliability(evidence_list)
    effective_lrs = [e["effective_lr"] for e in adjusted_evidence]
    
    # Step 4: 应用边际递减
    diminished_lrs = apply_diminishing_returns(effective_lrs)
    
    # Step 5: Odds-LR累乘
    posterior = _calculate_posterior(dynamic_prior, diminished_lrs)
    
    return {
        "posterior": posterior,
        "prior": base_prior,
        "dynamic_prior": dynamic_prior,
        "evidence_count": len(evidence_list),
        "effective_lrs": diminished_lrs,
        "reliability_adjusted": True,
        "evidence_details": adjusted_evidence,
    }
```

```python
from core.student_care_data_requirements import DATA_SOURCE_REQUIREMENTS

def detect_missing_evidence(
    dimension: str,
    signals: list[dict],
    profile: StudentCareProfile
) -> dict:
    """
    检测某维度的缺失证据
    
    Args:
        dimension: 维度标识（emotion/social/safety/family）
        signals: 学生信号列表
        profile: 学生画像
    
    Returns:
        {
            "covered_sources": [...],      # 已覆盖的数据源
            "missing_sources": [...],      # 缺失的数据源
            "coverage_ratio": float,       # 覆盖率
            "evidence_sufficient": bool,   # 证据是否充分
            "high_importance_missing": int # 高重要性缺失项数
        }
    """
    requirements = DATA_SOURCE_REQUIREMENTS.get(dimension, [])
    
    covered_sources = []
    missing_sources = []
    high_importance_missing = 0
    
    for req in requirements:
        matched = req["match_function"](signals, profile)
        
        source_info = {
            "id": req["id"],
            "label": req["label"],
            "description": req["description"],
            "importance": req["importance"],
        }
        
        if matched:
            covered_sources.append(source_info)
        else:
            missing_sources.append(source_info)
            if req["importance"] == "high":
                high_importance_missing += 1
    
    coverage_ratio = len(covered_sources) / len(requirements) if requirements else 0
    
    # 证据充分性判定：覆盖率>=60% 且 高重要性缺失项<=1
    evidence_sufficient = coverage_ratio >= 0.6 and high_importance_missing <= 1
    
    return {
        "covered_sources": covered_sources,
        "missing_sources": missing_sources,
        "coverage_ratio": round(coverage_ratio, 2),
        "evidence_sufficient": evidence_sufficient,
        "high_importance_missing": high_importance_missing,
    }
```

```python
def calculate_posterior_with_uncertainty(
    dimension: str,
    base_prior: float,
    linear_score: float,
    evidence_list: list[dict],
    signals: list[dict],
    profile: StudentCareProfile
) -> dict:
    """
    计算后验概率及置信区间（考虑缺失证据）
    
    Args:
        dimension: 维度标识
        base_prior: 基础先验
        linear_score: 线性分数
        evidence_list: 证据列表
        signals: 信号列表
        profile: 学生画像
    
    Returns:
        {
            "posterior": float,               # 后验概率（点估计）
            "confidence_interval": {          # 置信区间
                "lower": float,
                "upper": float,
            },
            "confidence": float,              # 置信度（0-1）
            "missing_evidence": dict,         # 缺失证据信息
            "evidence_sufficient": bool,      # 证据充分性
            "warning": str,                   # 警告信息
        }
    """
    # 1. 计算后验概率
    posterior_result = _calculate_posterior_for_dimension(
        dimension, base_prior, linear_score, evidence_list, profile
    )
    posterior = posterior_result["posterior"]
    
    # 2. 检测缺失证据
    missing_info = detect_missing_evidence(dimension, signals, profile)
    
    # 3. 计算置信区间宽度
    # 基础宽度：证据数量越少，区间越宽
    evidence_count = len(evidence_list)
    if evidence_count >= 5:
        base_interval_width = 0.08
    elif evidence_count >= 3:
        base_interval_width = 0.15
    else:
        base_interval_width = 0.25
    
    # 缺失惩罚：每缺失一个高重要性数据源，增加区间宽度
    missing_penalty = missing_info["high_importance_missing"] * 0.05
    total_interval_width = base_interval_width + missing_penalty
    
    # 4. 计算置信度
    base_confidence = 0.85 - (total_interval_width - 0.08) * 0.5
    confidence = max(0.42, min(base_confidence, 0.93))
    
    # 5. 生成置信区间
    lower_bound = max(0, posterior - total_interval_width)
    upper_bound = min(1, posterior + total_interval_width)
    
    # 6. 生成警告信息
    if not missing_info["evidence_sufficient"]:
        missing_labels = [s["label"] for s in missing_info["missing_sources"][:2]]
        warning = f"证据充分度不足，缺少：{', '.join(missing_labels)}"
    else:
        warning = None
    
    return {
        "posterior": posterior,
        "confidence_interval": {
            "lower": round(lower_bound, 3),
            "upper": round(upper_bound, 3),
        },
        "confidence": round(confidence, 3),
        "missing_evidence": missing_info,
        "evidence_sufficient": missing_info["evidence_sufficient"],
        "warning": warning,
        "evidence_details": posterior_result["evidence_details"],
    }
```

### B2. 证据冲突消解机制

#### 问题分析

当风险证据与保护性证据同时存在时，简单累乘可能导致不合理结果：

```

- 风险证据1：行为冲突事件 LR=2.2
- 风险证据2：图谱冲突共现 LR=1.9
- 保护性证据1：教师确认已解决 LR=0.5
- 保护性证据2：存在同伴支持 LR=0.55

当前处理：2.2 × 1.9 × 0.5 × 0.55 = 1.15
```

**2. 实现冲突消解函数**

```python
# 修改文件：backend/services/student_care_bayes_service.py

import math
from core.student_care_conflict_resolution import (
    EVIDENCE_GROUPING_STRATEGY,
    CONFLICT_RESOLUTION_RULES
)

def group_evidence_by_type(evidence_list: list[dict]) -> dict:
    """
    将证据分为风险组与保护组
    
    Returns:
        {
            "risk_evidence": [...],        # LR > 1 的证据
            "protective_evidence": [...],  # LR < 1 的证据
            "neutral_evidence": [...],     # LR = 1 的证据（通常无）
        }
    """
    risk_evidence = []
    protective_evidence = []
    neutral_evidence = []
    
    for evidence in evidence_list:
        lr = evidence.get("effective_lr", evidence.get("lr", 1.0))
        if lr > 1.0:
            risk_evidence.append(evidence)
        elif lr < 1.0:
            protective_evidence.append(evidence)
        else:
            neutral_evidence.append(evidence)
    
    return {
        "risk_evidence": risk_evidence,
        "protective_evidence": protective_evidence,
        "neutral_evidence": neutral_evidence,
    }


def combine_evidence_group(
    evidence_group: list[dict],
    diminishing_method: str = "sqrt"
) -> float:
    """
    合并单组证据（应用边际递减）
    
    Args:
        evidence_group: 证据列表
        diminishing_method: 边际递减方法
    
    Returns:
        float: 组合后的似然比
    """
    if not evidence_group:
        return 1.0  # 无证据时，中性
    
    # 提取有效似然比
    lrs = [e.get("effective_lr", e.get("lr", 1.0)) for e in evidence_group]
    
    # 应用边际递减
    diminished_lrs = apply_diminishing_returns(lrs, method=diminishing_method)
    
    # 累乘
    combined_lr = math.prod(diminished_lrs)
    
    return combined_lr


def resolve_evidence_conflict(
    risk_lr: float,
    protective_lr: float,
    evidence_count: int
) -> tuple[float, str]:
    """
    解决风险证据与保护性证据的冲突
    
    Args:
        risk_lr: 风险组组合似然比
        protective_lr: 保护组组合似然比
        evidence_count: 总证据数量
    
    Returns:
        tuple[float, str]: (最终似然比, 冲突消解说明)
    
    Example:
        >>> resolve_evidence_conflict(2.5, 0.4, 4)
        (1.0, "保护性证据占优，风险降低至中性")
    """
    # 计算双方强度
    risk_strength = risk_lr - 1.0 if risk_lr > 1 else 0
    protective_strength = 1.0 - protective_lr if protective_lr < 1 else 0
    
    # 判断冲突类型
    if protective_strength > risk_strength * 1.2:
        # 保护性证据显著更强
        # 最终LR = 1 / (risk_lr * protective_lr)，但设置下限
        final_lr = max(1.0 / (risk_lr * protective_lr), 0.5)
        reason = "保护性证据占优，风险大幅降低"
        
    elif risk_strength > protective_strength * 1.2:
        # 风险证据显著更强
        # 风险LR被保护性证据部分抵消
        mitigation_factor = 1.0 - (protective_strength / (risk_strength + protective_strength)) * 0.4
        final_lr = risk_lr ** mitigation_factor
        reason = "风险证据占优，但保护性因素部分缓解"
        
    else:
        # 双方均衡
        # 采用几何平均，略偏保守（risk优先）
        final_lr = math.sqrt(risk_lr * (1.0 / protective_lr)) * 1.1
        final_lr = max(0.8, min(final_lr, 1.3))
        reason = "风险与保护证据均衡，保持谨慎提示"
    
    # 证据数量修正：证据少时更保守
    if evidence_count <= 2:
        final_lr = max(final_lr * 1.15, 1.0)  # 证据少时提高风险提示
    
    return final_lr, reason


def calculate_posterior_with_conflict_resolution(
    base_prior: float,
    evidence_list: list[dict],
    dimension: str = None
) -> dict:
    """
    计算后验概率（带冲突消解）
    
    流程：
    1. 证据分组（风险 vs 保护）
    2. 组内合并（应用边际递减）
    3. 冲突消解
    4. 计算后验概率
    
    Returns:
        {
            "posterior": float,
            "prior": float,
            "risk_group_lr": float,
            "protective_group_lr": float,
            "final_lr": float,
            "conflict_resolution": str,
            "group_details": dict,
        }
    """
    # Step 1: 证据分组
    grouped = group_evidence_by_type(evidence_list)
    
    # Step 2: 组内合并
    risk_lr = combine_evidence_group(
        grouped["risk_evidence"],
        EVIDENCE_GROUPING_STRATEGY["risk_group"]["diminishing_method"]
    )
    
    protective_lr = combine_evidence_group(
        grouped["protective_evidence"],
        EVIDENCE_GROUPING_STRATEGY["protective_group"]["diminishing_method"]
    )
    
    # Step 3: 冲突消解
    final_lr, conflict_reason = resolve_evidence_conflict(
        risk_lr,
        protective_lr,
        len(evidence_list)
    )
    
    # Step 4: 计算后验概率
    # Odds = Prior / (1 - Prior)
    odds = base_prior / (1 - base_prior)
    # Odds' = Odds × Final_LR
    updated_odds = odds * final_lr
    # Posterior = Odds' / (1 + Odds')
    posterior = updated_odds / (1 + updated_odds)
    
    return {
        "posterior": posterior,
        "prior": base_prior,
        "risk_group_lr": risk_lr,
        "protective_group_lr": protective_lr,
        "final_lr": final_lr,
        "conflict_resolution": conflict_reason,
        "group_details": {
            "risk_count": len(grouped["risk_evidence"]),
            "protective_count": len(grouped["protective_evidence"]),
            "risk_evidence": [e["key"] for e in grouped["risk_evidence"]],
            "protective_evidence": [e["key"] for e in grouped["protective_evidence"]],
        }
    }
```

### B5. 似然比自适应学习

#### 问题分析

当前所有证据的似然比都是静态配置，无法从教师确认历史中学习：

```

问题：该证据的LR仍保持高值，继续导致误判

场景2：某证据在历史判断中多次被确认为真风险
问题：该证据的LR未得到强化，可能低估风险
```

#### 技术方案

**1. 定义学习策略**

**2. 实现MAB优化器**

```python
# 新增文件：backend/services/student_care_mab_service.py

import math
from datetime import datetime
from typing import Optional
from sqlalchemy.orm import Session
from models.student_care import StudentCareBayesRule, StudentCareAgentRecord
from core.student_care_mab_config import MAB_OPTIMIZATION_CONFIG, UCB_PARAMS

class LikelihoodRatioOptimizer:
    """基于多臂赌博机的似然比优化器"""
    
    def __init__(self, db: Session):
        self.db = db
        self.config = MAB_OPTIMIZATION_CONFIG
    
    def get_current_lr(self, dimension: str, evidence_key: str) -> float:
        """获取当前似然比（考虑版本）"""
        rule = self.db.query(StudentCareBayesRule).filter(
            StudentCareBayesRule.dimension == dimension,
            StudentCareBayesRule.evidence_key == evidence_key,
            StudentCareBayesRule.enabled == True
        ).first()
        
        return rule.likelihood_ratio if rule else 1.0
    
    def update_lr_from_teacher_review(
        self,
        dimension: str,
        evidence_key: str,
        teacher_decision: str,
        confidence: float = 1.0
    ) -> dict:
        """
        基于教师确认更新似然比
        
        Args:
            dimension: 维度
            evidence_key: 证据标识
            teacher_decision: "true_risk" 或 "false_alarm"
            confidence: 教师置信度（1-5）
        
        Returns:
            {
                "updated": bool,
                "old_lr": float,
                "new_lr": float,
                "reason": str,
            }
        """
        # 获取当前LR
        current_lr = self.get_current_lr(dimension, evidence_key)
        
        # 检查是否有足够样本
        stats = self._get_evidence_statistics(dimension, evidence_key)
        if stats["total_count"] < self.config["update_trigger"]["min_samples"]:
            return {
                "updated": False,
                "old_lr": current_lr,
                "new_lr": current_lr,
                "reason": f"样本不足（{stats['total_count']}/{self.config['update_trigger']['min_samples']}）",
            }
        
        # 检查教师一致性
        if teacher_decision == "true_risk":
            consistency = stats["true_risk_rate"]
        else:
            consistency = stats["false_alarm_rate"]
        
        if consistency < self.config["update_trigger"]["consistency_threshold"]:
            return {
                "updated": False,
                "old_lr": current_lr,
                "new_lr": current_lr,
                "reason": f"教师一致性不足（{consistency:.2%}）",
            }
        
        # 计算新LR
        if teacher_decision == "true_risk":
            adjustment = self.config["learning_rate"]["true_risk"]
        else:
            adjustment = self.config["learning_rate"]["false_alarm"]
        
        # 考虑教师置信度
        confidence_factor = confidence / 3.0  # 归一化到[0.33, 1.67]
        adjustment = 1.0 + (adjustment - 1.0) * confidence_factor
        
        new_lr = current_lr * adjustment
        
        # 约束到合理范围
        if current_lr >= 1.0:
            # 风险证据
            bounds = self.config["lr_bounds"]["risk_evidence"]
            new_lr = max(bounds[0], min(new_lr, bounds[1]))
        else:
            # 保护性证据
            bounds = self.config["lr_bounds"]["protective_evidence"]
            new_lr = max(bounds[0], min(new_lr, bounds[1]))
        
        # 保存更新
        self._save_lr_update(dimension, evidence_key, current_lr, new_lr, teacher_decision)
        
        return {
            "updated": True,
            "old_lr": current_lr,
            "new_lr": round(new_lr, 3),
            "reason": f"教师一致性{consistency:.2%}，应用{teacher_decision}调整",
        }
    
    def _get_evidence_statistics(self, dimension: str, evidence_key: str) -> dict:
        """获取证据的统计信息"""
        # 查询该证据在研判记录中的历史
        # 实现省略，返回统计结果
        return {
            "total_count": 10,
            "true_risk_count": 7,
            "false_alarm_count": 3,
            "true_risk_rate": 0.7,
            "false_alarm_rate": 0.3,
        }
    
    def _save_lr_update(
        self,
        dimension: str,
        evidence_key: str,
        old_lr: float,
        new_lr: float,
        trigger: str
    ):
        """保存LR更新记录"""
        # 更新数据库
        rule = self.db.query(StudentCareBayesRule).filter(
            StudentCareBayesRule.dimension == dimension,
            StudentCareBayesRule.evidence_key == evidence_key
        ).first()
        
        if rule:
            rule.likelihood_ratio = new_lr
            rule.version += 1
            rule.last_updated = datetime.now()
            rule.update_reason = trigger
            self.db.commit()
    
    def calculate_ucb_score(
        self,
        dimension: str,
        evidence_key: str
    ) -> float:
        """
        计算UCB分数（探索价值）
        
        UCB = Average_Reward + c * sqrt(ln(N) / n)
        
        其中：
        - Average_Reward: 该证据的平均收益（正确判断率）
        - c: 置信度参数
        - N: 总试验次数
        - n: 该证据的试验次数
        """
        stats = self._get_evidence_statistics(dimension, evidence_key)
        
        if stats["total_count"] < UCB_PARAMS["min_samples_for_ucb"]:
            # 样本不足，给高UCB鼓励探索
            return float('inf')
        
        # 平均收益 = 正确判断率
        avg_reward = stats["true_risk_rate"] if stats["true_risk_count"] > stats["false_alarm_count"] else stats["false_alarm_rate"]
        
        # UCB项
        total_trials = sum([
            self._get_evidence_statistics(dimension, key)["total_count"]
            for key in self._get_all_evidence_keys(dimension)
        ])
        
        ucb_term = UCB_PARAMS["confidence_level"] * math.sqrt(
            math.log(total_trials) / stats["total_count"]
        )
        
        return avg_reward + ucb_term
    
    def _get_all_evidence_keys(self, dimension: str) -> list[str]:
        """获取某维度的所有证据键"""
        rules = self.db.query(StudentCareBayesRule.evidence_key).filter(
            StudentCareBayesRule.dimension == dimension
        ).all()
        return [r[0] for r in rules]
```

```python
            profile_data: 学生画像数据（含时间信息）
        
        Returns:
            List[Dict]: 异常列表，每项包含 {
                "evidence_key": str,
                "anomaly_type": str,
                "severity": str,  # "high"/"medium"/"low"
                "description": str,
                "suggestion": str,
            }
        """
        anomalies = []
        
        for evidence in evidence_list:
            # 检测规则1：与整体趋势矛盾
            if self.rules["contradict_trend"]["enabled"]:
                anomaly = self._check_contradict_trend(
                    evidence, overall_risk_direction
                )
                if anomaly:
                    anomalies.append(anomaly)
            
            # 检测规则2：证据过时
            if self.rules["evidence_outdated"]["enabled"]:
                anomaly = self._check_evidence_outdated(
                    evidence, profile_data
                )
                if anomaly:
                    anomalies.append(anomaly)
            
            # 检测规则3：证据质量可疑
            if self.rules["low_quality_source"]["enabled"]:
                anomaly = self._check_low_quality_source(evidence)
                if anomaly:
                    anomalies.append(anomaly)
        
        # 检测规则4：孤立强证据
        if self.rules["isolated_strong_evidence"]["enabled"]:
            anomaly = self._check_isolated_strong_evidence(evidence_list)
            if anomaly:
                anomalies.append(anomaly)
        
        return anomalies
    
    def _check_contradict_trend(
        self,
        evidence: Dict,
        overall_risk_direction: str
    ) -> Optional[Dict]:
        """检测与整体趋势矛盾"""
        lr = evidence.get("effective_lr", evidence.get("lr", 1.0))
        evidence_direction = "risk" if lr > 1.0 else "protective"
        
        # 如果证据方向与整体趋势相反，且偏离较大
        if evidence_direction != overall_risk_direction:
            deviation = abs(lr - 1.0)
            if deviation > self.rules["contradict_trend"]["threshold"]:
                severity = "high" if deviation > 2.0 else "medium"
                
                return {
                    "evidence_key": evidence["key"],
                    "anomaly_type": "contradict_trend",
                    "severity": severity,
                    "description": f"证据方向（{evidence_direction}）与整体趋势（{overall_risk_direction}）矛盾",
                    "suggestion": "建议教师人工确认该证据的有效性",
                    "lr": lr,
                    "direction": evidence_direction,
                }
        
        return None
    
    def _check_evidence_outdated(
        self,
        evidence: Dict,
        profile_data: Dict
    ) -> Optional[Dict]:
        """检测证据过时"""
        source = evidence.get("source", "")
        
        # 某些来源需要检查时效性
        time_sensitive_sources = ["behavior_event", "care_observation", "attendance"]
        
        if source in time_sensitive_sources:
            evidence_time = evidence.get("occurred_at")
            if evidence_time:
                age_days = (datetime.now() - evidence_time).days
                max_age = self.rules["evidence_outdated"]["max_age_days"]
                
                if age_days > max_age:
                    return {
                        "evidence_key": evidence["key"],
                        "anomaly_type": "evidence_outdated",
                        "severity": "medium",
                        "description": f"证据已过时（{age_days}天前），可能不再反映当前状态",
                        "suggestion": "建议确认该证据是否仍有参考价值",
                        "age_days": age_days,
                        "max_age": max_age,
                    }
        
        return None
    
    def _check_low_quality_source(self, evidence: Dict) -> Optional[Dict]:
        """检测证据质量可疑"""
        reliability = evidence.get("reliability", 0.65)
        quality_threshold = self.rules["low_quality_source"]["quality_threshold"]
        
        if reliability < quality_threshold:
            return {
                "evidence_key": evidence["key"],
                "anomaly_type": "low_quality_source",
                "severity": "low",
                "description": f"证据来源可靠性较低（{reliability:.2f}），可能导致误判",
                "suggestion": "建议参考其他更可靠的证据",
                "reliability": reliability,
            }
        
        return None
    
    def _check_isolated_strong_evidence(
        self,
        evidence_list: List[Dict]
    ) -> Optional[Dict]:
        """检测孤立强证据"""
        strong_evidence = [
            e for e in evidence_list
            if abs(e.get("effective_lr", e.get("lr", 1.0)) - 1.0) >= 
               self.rules["isolated_strong_evidence"]["lr_threshold"] - 1.0
        ]
        
        if len(strong_evidence) == 1:
            # 只有一个强证据，检查是否有足够支撑
            supporting_count = len(evidence_list) - 1
            min_support = self.rules["isolated_strong_evidence"]["min_supporting_evidence"]
            
            if supporting_count < min_support:
                return {
                    "evidence_key": strong_evidence[0]["key"],
                    "anomaly_type": "isolated_strong_evidence",
                    "severity": "medium",
                    "description": f"单个强证据缺乏支撑（仅{supporting_count}个辅助证据）",
                    "suggestion": "建议收集更多证据后再做判断",
                    "supporting_count": supporting_count,
                    "min_required": min_support,
                }
        
        return None
```
